```python
import numpy as np
import matplotlib.pyplot as plt
import pywt # 需要安装 PyWavelets 库
import os

# --- 1. Mac 环境字体配置 (陈教授专用配置) ---
plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'Heiti TC', 'sans-serif']
plt.rcParams['axes.unicode_minus'] = False

# --- 2. 信号处理逻辑 ---

# 生成一个扫频信号 (Chirp Signal)
# 采样率 1000Hz, 持续 1秒
fs = 1000
t = np.linspace(0, 1, fs, endpoint=False)
# 频率从 20Hz 线性增加到 200Hz
# sin(2*pi * (f_start + (f_end-f_start)/2/T * t) * t)
f_start = 20
f_end = 200
signal = np.sin(2 * np.pi * (f_start + (f_end - f_start) / 2 * t) * t)

# (1) 计算 FFT (对比用)
N = len(signal)
freqs = np.fft.fftfreq(N, 1/fs)
fft_vals = np.abs(np.fft.fft(signal)) / N

# (2) 计算连续小波变换 (CWT)
# 选择一个小波基，例如 Complex Morlet ('cmor1.5-1.0')
wavelet_name = 'cmor1.5-1.0'
# 定义尺度范围
scales = np.arange(1, 128)
# 计算 CWT
# coefficients: 小波系数 (复数), frequencies: 对应的物理频率
coefficients, frequencies = pywt.cwt(signal, scales, wavelet_name, sampling_period=1/fs)
power = (np.abs(coefficients)) ** 2

# --- 3. 绘图 ---
fig = plt.figure(figsize=(12, 10))

# 子图1: 时域波形
ax1 = fig.add_subplot(3, 1, 1)
ax1.plot(t, signal)
ax1.set_title('时域信号：扫频信号 (20Hz -> 200Hz)')
ax1.set_xlabel('时间 (s)')
ax1.set_ylabel('幅度')
ax1.grid(True, alpha=0.3)

# 子图2: FFT 频谱
ax2 = fig.add_subplot(3, 1, 2)
# 只画正频率部分
mask = freqs > 0
ax2.plot(freqs[mask], fft_vals[mask], color='orange')
ax2.set_title('FFT 频谱：只知道有哪些频率，不知道发生的时间')
ax2.set_xlabel('频率 (Hz)')
ax2.set_ylabel('幅度')
ax2.set_xlim(0, 300)
ax2.grid(True, alpha=0.3)

# 子图3: 小波变换时频图 (Scaleogram)
ax3 = fig.add_subplot(3, 1, 3)
# 绘制等高线图
im = ax3.contourf(t, frequencies, power, levels=50, cmap='jet')
ax3.set_title('小波变换时频图：清晰展示频率随时间的变化')
ax3.set_ylabel('频率 (Hz)')
ax3.set_xlabel('时间 (s)')
# 调整 y 轴显示范围
ax3.set_ylim(0, 300)
fig.colorbar(im, ax=ax3, label='能量')

plt.tight_layout()

# --- 4. 自动保存与显示 (固定配置) ---
filename = "小波变换时频分析演示.png"

try:
    if os.path.exists(full_path):
        os.remove(full_path) # 覆盖旧图
    plt.savefig(full_path, dpi=300, bbox_inches='tight')
    plt.close() # 释放内存
    print(f"![[{filename}]]") # 触发 Obsidian 显示
except Exception as e:
    print(f"Error: {e}")
```
