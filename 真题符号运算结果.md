```python
import numpy as np
import matplotlib.pyplot as plt
import sympy as sp
import os

# --- 1. Mac 环境字体配置 ---
plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'Heiti TC', 'sans-serif']
plt.rcParams['axes.unicode_minus'] = False

# --- 2. 信号处理逻辑：符号运算验证双线性变换 ---

def symbolic_bilinear_transform():
    # 定义符号变量
    z, T = sp.symbols('z T')
    s = (2/T) * (1 - z**-1) / (1 + z**-1)
    
    # 案例 1: 2017年 Q14, H(s) = 1/s
    Hs_1 = 1/s
    Hz_1 = sp.simplify(Hs_1)
    
    # 案例 2: 2022年 Q6, H(s) = 2 / (s^2 + 3s + 2), T=2
    # 注意：此处 T=2
    s_val = (1 - z**-1) / (1 + z**-1) # 2/T = 1 when T=2
    Hs_2 = 2 / (s_val**2 + 3*s_val + 2)
    Hz_2 = sp.simplify(Hs_2)
    
    return Hz_1, Hz_2

# 执行符号计算
res1, res2 = symbolic_bilinear_transform()

# --- 3. 结果可视化 (生成一张包含公式的图片) ---
filename = "真题符号运算结果.png"
vault_path = "/Users/heyuhang/Documents/Obsidian Vault"
full_path = os.path.join(vault_path, filename)

fig, ax = plt.subplots(figsize=(10, 4))
ax.axis('off')

# 将 Latex 字符串放入图表
text_str = (
    r"$\bf{2017\ Q14:}\ H(s) = \frac{1}{s}$" + "\n" +
    r"$\rightarrow H(z) = " + sp.latex(res1) + r"$" + "\n\n" +
    r"$\bf{2022\ Q6:}\ H(s) = \frac{2}{s^2+3s+2}, T=2$" + "\n" +
    r"$\rightarrow H(z) = " + sp.latex(res2) + r"$"
)

ax.text(0.05, 0.5, text_str, fontsize=16, va='center')

try:
    if os.path.exists(full_path):
        os.remove(full_path)
    plt.savefig(full_path, dpi=300, bbox_inches='tight')
    plt.close()
    print(f"![[{filename}]]")
except Exception as e:
    print(f"Error: {e}")
```
