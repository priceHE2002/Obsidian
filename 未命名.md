```python
import numpy as np
import matplotlib.pyplot as plt
import os

# --- 1. Mac 环境字体配置 (陈教授专用配置) ---
plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'Heiti TC', 'sans-serif']
plt.rcParams['axes.unicode_minus'] = False

# --- 2. 信号处理逻辑 ---

# 生成一个随机的实数序列
N = 8
k = np.arange(N)
# 为了演示方便，我们固定一个实序列
x = np.array([1, 2, -1, 3, 0, -2, 4, 1], dtype=float)

# 计算 DFT
X = np.fft.fft(x)

# 验证共轭对称性：X[m] 与 X[N-m] (即 X[-m]) 应该是共轭关系
# 我们取 m=1 和 m=7 (N-1) 来观察
m_idx = 1
m_sym_idx = N - m_idx

val_1 = X[m_idx]
val_2 = X[m_sym_idx]

print(f"实序列 x[k]: {x}")
print(f"X[{m_idx}] = {val_1:.2f}")
print(f"X[{m_sym_idx}] = {val_2:.2f}")
print(f"验证共轭: X[{m_idx}] == conj(X[{m_sym_idx}]) ? {np.isclose(val_1, np.conj(val_2))}")

# --- 3. 绘图：可视化这种对称性 ---
# 实部应该是偶对称，虚部应该是奇对称
fig, ax = plt.subplots(2, 1, figsize=(10, 8))

# 绘制实部
ax[0].stem(k, np.real(X), basefmt=" ", label='DFT 实部')
ax[0].set_title(f'实序列 DFT 的实部 (偶对称)')
# 标注对称点
ax[0].plot([m_idx, m_sym_idx], [np.real(val_1), np.real(val_2)], 'ro', markerfacecolor='none', markersize=10)
ax[0].annotate('', xy=(m_idx, np.real(val_1)), xytext=(m_sym_idx, np.real(val_2)),
               arrowprops=dict(arrowstyle="<->", color='red', lw=1.5, connectionstyle="arc3,rad=-0.3"))
ax[0].grid(True, alpha=0.3)
ax[0].legend()

# 绘制虚部
ax[1].stem(k, np.imag(X), basefmt=" ", linefmt='C1-', markerfmt='C1o', label='DFT 虚部')
ax[1].set_title(f'实序列 DFT 的虚部 (奇对称)')
# 标注反对称点
ax[1].plot([m_idx, m_sym_idx], [np.imag(val_1), np.imag(val_2)], 'bo', markerfacecolor='none', markersize=10)
ax[1].annotate('', xy=(m_idx, np.imag(val_1)), xytext=(m_sym_idx, np.imag(val_2)),
               arrowprops=dict(arrowstyle="<->", color='blue', lw=1.5, connectionstyle="arc3,rad=0.3"))
ax[1].text(N/2, 0, "互为相反数", ha='center', color='blue')
ax[1].grid(True, alpha=0.3)
ax[1].legend()

plt.tight_layout()

# --- 4. 自动保存与显示 (固定配置) ---
filename = "实信号DFT共轭对称性.png"

        os.remove(full_path) # 覆盖旧图
    plt.savefig(full_path, dpi=300, bbox_inches='tight')
    plt.close() # 释放内存
    print(f"![[{filename}]]") # 触发 Obsidian 显示
except Exception as e:
    print(f"Error: {e}")
```
