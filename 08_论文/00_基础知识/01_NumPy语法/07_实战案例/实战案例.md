# 7. 实战案例：纯 NumPy 两层 MLP 训练 MNIST

用本章学到的所有 NumPy 知识，从零搭建并训练一个两层全连接网络做 MNIST 手写数字分类。所有操作——包括前向传播、反向传播、BN、SGD——全部用 NumPy 实现，不依赖任何深度学习框架。

## 7.1 整体架构

```
输入 (784) → FC1 (784→256) → BN → ReLU → FC2 (256→10) → Softmax + CE Loss
```

## 7.2 完整代码

```python
import numpy as np
from sklearn.datasets import fetch_openml
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler

# ==================== 数据准备 ====================

print("加载 MNIST...")
X, y = fetch_openml('mnist_784', version=1, return_X_y=True, as_frame=False)
X = X.astype(np.float32) / 255.0   # 归一化到 [0, 1]
y = y.astype(np.int32)

# 划分训练集和验证集
X_train, X_val, y_train, y_val = train_test_split(
    X, y, test_size=0.1, random_state=42
)

print(f"训练集: {X_train.shape[0]}, 验证集: {X_val.shape[0]}")
print(f"输入维度: {X_train.shape[1]}, 类别数: 10")

# ==================== 网络层定义 ====================

class Linear_Numpy:
    """全连接层"""
    def __init__(self, in_features, out_features):
        # He 初始化（适合 ReLU）
        self.W = np.random.randn(in_features, out_features) * np.sqrt(2.0 / in_features)
        self.b = np.zeros(out_features)

    def forward(self, x):
        self.x = x  # 缓存用于反向传播
        return x @ self.W + self.b

    def backward(self, dout):
        # dout: (N, out_features)
        self.dW = self.x.T @ dout          # (in, out)
        self.db = np.sum(dout, axis=0)      # (out,)
        dx = dout @ self.W.T                # (N, in)
        return dx

    def step(self, lr):
        self.W -= lr * self.dW
        self.b -= lr * self.db


class BatchNorm1d_Numpy:
    """一维批量归一化"""
    def __init__(self, num_features, eps=1e-5, momentum=0.1):
        self.gamma = np.ones(num_features)
        self.beta = np.zeros(num_features)
        self.eps = eps
        self.momentum = momentum
        self.running_mean = np.zeros(num_features)
        self.running_var = np.ones(num_features)

    def forward(self, x, training=True):
        N = x.shape[0]

        if training:
            self.mu = np.mean(x, axis=0)
            self.var = np.var(x, axis=0)
            self.x_centered = x - self.mu
            self.std_inv = 1.0 / np.sqrt(self.var + self.eps)
            self.x_normalized = self.x_centered * self.std_inv

            self.running_mean = (1 - self.momentum) * self.running_mean \
                                + self.momentum * self.mu
            self.running_var = (1 - self.momentum) * self.running_var \
                               + self.momentum * self.var
        else:
            self.x_normalized = (x - self.running_mean) / np.sqrt(self.running_var + self.eps)

        return self.gamma * self.x_normalized + self.beta

    def backward(self, dout):
        N = dout.shape[0]

        self.dgamma = np.sum(dout * self.x_normalized, axis=0)
        self.dbeta = np.sum(dout, axis=0)

        dx_norm = dout * self.gamma

        # BN 反向传播核心公式
        dx = (1.0 / N) * self.std_inv * (
            N * dx_norm
            - np.sum(dx_norm, axis=0)
            - self.x_normalized * np.sum(dx_norm * self.x_normalized, axis=0)
        )
        return dx

    def step(self, lr):
        self.gamma -= lr * self.dgamma
        self.beta -= lr * self.dbeta


class ReLU_Numpy:
    """ReLU 激活"""
    def forward(self, x):
        self.mask = x > 0
        return x * self.mask

    def backward(self, dout):
        return dout * self.mask


# ==================== Softmax + CrossEntropy ====================

def softmax_ce_loss(logits, targets):
    """组合损失 + 对 logits 的梯度"""
    N = logits.shape[0]

    # 数值稳定 softmax
    shifted = logits - np.max(logits, axis=1, keepdims=True)
    exps = np.exp(shifted)
    probs = exps / np.sum(exps, axis=1, keepdims=True)

    # 损失
    correct_log_probs = -np.log(probs[np.arange(N), targets] + 1e-8)
    loss = np.mean(correct_log_probs)

    # 梯度
    grad = probs.copy()
    grad[np.arange(N), targets] -= 1
    grad /= N

    return loss, grad


# ==================== SGD 优化器 ====================

class SGD_Numpy:
    def __init__(self, params, lr=0.01, momentum=0.9, weight_decay=1e-4):
        self.params = params
        self.lr = lr
        self.momentum = momentum
        self.weight_decay = weight_decay
        self.velocities = [np.zeros_like(p['layer'].W) for p in params]

    def zero_grad(self):
        pass  # 每步 backward 自动覆盖梯度

    def step(self):
        for i, p in enumerate(self.params):
            layer = p['layer']

            # L2 正则
            grad_w = layer.dW + self.weight_decay * layer.W

            # Momentum
            self.velocities[i] = (self.momentum * self.velocities[i]
                                  + (1 - self.momentum) * grad_w)

            # 更新
            layer.W -= self.lr * self.velocities[i]
            layer.b -= self.lr * layer.db

            # BN 的参数也更新
            if isinstance(layer, BatchNorm1d_Numpy):
                layer.step(self.lr)


# ==================== 准确率计算 ====================

def accuracy(predictions, targets):
    return np.mean(predictions == targets)


# ==================== 训练与验证 ====================

def train_epoch(model_layers, X, y, batch_size, optimizer):
    """训练一个 epoch"""
    indices = np.random.permutation(len(X))
    total_loss = 0.0
    num_batches = 0

    for start in range(0, len(X), batch_size):
        batch_idx = indices[start:start + batch_size]
        x_batch = X[batch_idx]
        y_batch = y[batch_idx]

        # ── 前向传播 ──
        out = x_batch
        for layer in model_layers:
            if isinstance(layer, BatchNorm1d_Numpy):
                out = layer.forward(out, training=True)
            else:
                out = layer.forward(out)

        loss, grad = softmax_ce_loss(out, y_batch)
        total_loss += loss
        num_batches += 1

        # ── 反向传播 ──
        dout = grad
        for layer in reversed(model_layers):
            dout = layer.backward(dout)

        # ── 参数更新 ──
        optimizer.step()

    return total_loss / num_batches


def val_epoch(model_layers, X, y, batch_size):
    """验证"""
    indices = np.arange(len(X))
    total_loss = 0.0
    all_preds = []
    num_batches = 0

    for start in range(0, len(X), batch_size):
        batch_idx = indices[start:start + batch_size]
        x_batch = X[batch_idx]
        y_batch = y[batch_idx]

        # 前向（training=False：BN 用 running 统计）
        out = x_batch
        for layer in model_layers:
            if isinstance(layer, BatchNorm1d_Numpy):
                out = layer.forward(out, training=False)
            else:
                out = layer.forward(out)

        loss, _ = softmax_ce_loss(out, y_batch)
        total_loss += loss
        num_batches += 1

        preds = np.argmax(out, axis=1)
        all_preds.extend(preds == y_batch)

    acc = np.mean(all_preds)
    return total_loss / num_batches, acc


# ==================== 搭建模型 ====================

fc1 = Linear_Numpy(in_features=784, out_features=256)
bn1 = BatchNorm1d_Numpy(num_features=256)
relu = ReLU_Numpy()
fc2 = Linear_Numpy(in_features=256, out_features=10)

model_layers = [fc1, bn1, relu, fc2]

# 优化器：收集所有层的参数
params = [
    {'layer': fc1},
    {'layer': fc2},
]
optimizer = SGD_Numpy(params, lr=0.01, momentum=0.9, weight_decay=1e-4)


# ==================== 训练 ====================

batch_size = 128
num_epochs = 20
best_acc = 0.0

print(f"\n开始训练（{num_epochs} epochs）...")
print("-" * 55)

for epoch in range(1, num_epochs + 1):
    train_loss = train_epoch(model_layers, X_train, y_train,
                             batch_size, optimizer)
    val_loss, val_acc = val_epoch(model_layers, X_val, y_val, batch_size)

    if val_acc > best_acc:
        best_acc = val_acc

    if epoch % 5 == 0 or epoch == 1:
        print(f"Epoch {epoch:3d} | "
              f"Train Loss: {train_loss:.4f} | "
              f"Val Loss: {val_loss:.4f} | "
              f"Val Acc: {val_acc:.4f}")

print("-" * 55)
print(f"训练完成！最佳验证准确率: {best_acc:.4f}")


# ==================== 推理示例 ====================

def predict_single(model_layers, x):
    """单张图片推理"""
    out = x.reshape(1, -1)
    for layer in model_layers:
        if isinstance(layer, BatchNorm1d_Numpy):
            out = layer.forward(out, training=False)
        else:
            out = layer.forward(out)
    probs = np.exp(out - np.max(out, axis=1, keepdims=True))
    probs /= np.sum(probs, axis=1, keepdims=True)
    return np.argmax(out, axis=1)[0], np.max(probs)

# 试一张
sample_x = X_val[0]
sample_y = y_val[0]
pred_class, conf = predict_single(model_layers, sample_x)
print(f"\n推理测试: 真实标签={sample_y}, 预测={pred_class}, 置信度={conf:.4f}")


# ==================== 各类别准确率 ====================

print("\n各类别准确率:")
class_correct = np.zeros(10, dtype=int)
class_total = np.zeros(10, dtype=int)

for start in range(0, len(X_val), 512):
    batch_idx = np.arange(start, min(start + 512, len(X_val)))
    x_batch = X_val[batch_idx]
    y_batch = y_val[batch_idx]

    out = x_batch
    for layer in model_layers:
        if isinstance(layer, BatchNorm1d_Numpy):
            out = layer.forward(out, training=False)
        else:
            out = layer.forward(out)
    preds = np.argmax(out, axis=1)

    for i in range(len(y_batch)):
        label = y_batch[i]
        class_correct[label] += (preds[i] == label)
        class_total[label] += 1

for i in range(10):
    acc = class_correct[i] / class_total[i] * 100
    print(f"  数字 {i}: {acc:.2f}%")
```

## 7.3 关键设计说明

**为什么用 im2col 风格的矩阵乘法实现 Linear？**

```python
x @ self.W  # (N, in) @ (in, out) → (N, out)
```

这就是全连接层的本质——一个矩阵乘法加偏置。NumPy 下写反向传播的公式和 PyTorch 中 `nn.Linear` 内部的计算完全一致。

**BN 的反向传播为什么这么复杂？**

BN 的链式法则要经过四个步骤：`y = γ * x_hat + β`、`x_hat = (x - μ) / √(σ² + ε)`、`μ = mean(x)`、`σ² = var(x)`。对 `x` 的梯度公式经过了化简，但本质就是标准链式求导。理解这一过程对面试中"手推 BN"很有帮助。

**He 初始化**：

`np.sqrt(2.0 / fan_in)`——这个 `2` 是为了弥补 ReLU 截断负半轴造成的信息损失。若用 Sigmoid/Tanh 则换成 Xavier 初始化 `np.sqrt(1.0 / fan_in)`。

## 7.4 训练效果预期

在 MNIST（60k 训练 / 10k 测试，两层 MLP 784→256→10）下，纯 NumPy 实现通常能跑到验证准确率约 97-98%。关键是 He 初始化 + BN + Momentum SGD 三个要素缺一不可。

## 7.5 总结：从 NumPy 到 PyTorch 的跨越

| 维度 | NumPy（本案例） | PyTorch |
|---|---|---|
| 前向传播 | 手动调每个 layer.forward() | `model(x)` 自动串联 |
| 反向传播 | 手动调每个 layer.backward() | `loss.backward()` 自动微分 |
| 参数更新 | `optimizer.step()` 手动实现 | `optimizer.step()` 内置 |
| GPU | 不支持 | `.to('cuda')` |
| BN 推理 | `training=False` 手动切换 | `model.eval()` 自动处理 |
| 梯度清零 | 每条路径手动覆盖 | `optimizer.zero_grad()` |

理解了这份 NumPy 实现，就能真正理解 PyTorch 在"背后做了什么"。
