# 5. 训练流程

一个标准的 PyTorch 训练流程包括：定义模型、损失函数、优化器、训练 / 验证循环、指标记录和学习率调度。

## 5.1 标准训练模板

```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader

# --- 准备 ---
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
model = MyModel().to(device)
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=1e-3)

# --- 一个 epoch 的训练 ---
def train_epoch(model, dataloader, criterion, optimizer, device):
    model.train()
    total_loss = 0.0
    correct = 0
    total = 0

    for x, y in dataloader:
        x, y = x.to(device), y.to(device)

        # 1. 清零梯度
        optimizer.zero_grad()

        # 2. 前向传播
        output = model(x)

        # 3. 计算 loss
        loss = criterion(output, y)

        # 4. 反向传播
        loss.backward()

        # 5. 更新参数
        optimizer.step()

        # 统计
        total_loss += loss.item() * x.size(0)
        pred = output.argmax(dim=1)
        correct += (pred == y).sum().item()
        total += y.size(0)

    return total_loss / total, correct / total

# --- 一个 epoch 的验证 ---
@torch.no_grad()
def val_epoch(model, dataloader, criterion, device):
    model.eval()
    total_loss = 0.0
    correct = 0
    total = 0

    for x, y in dataloader:
        x, y = x.to(device), y.to(device)
        output = model(x)
        loss = criterion(output, y)

        total_loss += loss.item() * x.size(0)
        pred = output.argmax(dim=1)
        correct += (pred == y).sum().item()
        total += y.size(0)

    return total_loss / total, correct / total
```

## 5.2 完整训练循环

```python
num_epochs = 50
best_acc = 0.0

for epoch in range(1, num_epochs + 1):
    train_loss, train_acc = train_epoch(
        model, train_loader, criterion, optimizer, device
    )
    val_loss, val_acc = val_epoch(
        model, val_loader, criterion, device
    )

    # 学习率调度
    scheduler.step()  # 或 scheduler.step(val_loss)

    print(f"Epoch {epoch:3d} | "
          f"Train Loss: {train_loss:.4f} Acc: {train_acc:.4f} | "
          f"Val Loss: {val_loss:.4f} Acc: {val_acc:.4f}")

    # 保存最佳模型
    if val_acc > best_acc:
        best_acc = val_acc
        torch.save(model.state_dict(), 'best_model.pth')
        print(f"  → 保存最佳模型 (acc={best_acc:.4f})")

print(f"训练完成，最佳验证准确率: {best_acc:.4f}")
```

## 5.3 常用损失函数

```python
import torch.nn as nn

# 分类
ce = nn.CrossEntropyLoss()                       # 多分类（内置 softmax）
bce = nn.BCEWithLogitsLoss()                     # 二分类（内置 sigmoid，比 BCE+手动 sigmoid 数值更稳定）

# 回归
mse = nn.MSELoss()                               # 均方误差
l1 = nn.L1Loss()                                 # 绝对误差
huber = nn.SmoothL1Loss()                        # 结合 MSE 和 L1 优点（对离群点鲁棒）

# 自定义损失函数
class FocalLoss(nn.Module):
    def __init__(self, gamma=2.0, alpha=0.25):
        super().__init__()
        self.gamma = gamma
        self.alpha = alpha

    def forward(self, inputs, targets):
        ce_loss = nn.functional.cross_entropy(inputs, targets, reduction='none')
        pt = torch.exp(-ce_loss)
        focal_loss = self.alpha * (1 - pt) ** self.gamma * ce_loss
        return focal_loss.mean()
```

## 5.4 学习率调度

```python
from torch.optim.lr_scheduler import (
    StepLR, MultiStepLR, CosineAnnealingLR,
    ReduceLROnPlateau, OneCycleLR
)

optimizer = optim.SGD(model.parameters(), lr=0.1)

# 每 N 步衰减
scheduler1 = StepLR(optimizer, step_size=30, gamma=0.1)

# 在指定 epoch 衰减
scheduler2 = MultiStepLR(optimizer, milestones=[30, 60, 80], gamma=0.1)

# 余弦退火（CV 常用）
scheduler3 = CosineAnnealingLR(optimizer, T_max=num_epochs)

# 监控验证集指标自动降低（val loss 不降时减 lr）
scheduler4 = ReduceLROnPlateau(optimizer, mode='min', patience=5, factor=0.5)

# OneCycleLR（训练前期升 lr，后期降 lr，收敛快）
scheduler5 = OneCycleLR(optimizer, max_lr=0.1,
                        steps_per_epoch=len(train_loader),
                        epochs=num_epochs)
```

**调用时机**：

- `StepLR` / `CosineAnnealingLR`：每个 epoch 结束后调用 `scheduler.step()`。
- `ReduceLROnPlateau`：每个 epoch 结束后调用 `scheduler.step(val_loss)`（需传监控值）。
- `OneCycleLR`：每个 batch 结束后调用 `scheduler.step()`。

## 5.5 梯度裁剪

```python
# 防止梯度爆炸（RNN / Transformer 训练常用）
torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)

# 放在 loss.backward() 之后，optimizer.step() 之前
loss.backward()
torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=5.0)
optimizer.step()
```

## 5.6 混合精度训练（加速 + 省显存）

```python
scaler = torch.amp.GradScaler('cuda')  # 或 torch.cuda.amp.GradScaler()

for x, y in dataloader:
    x, y = x.to(device), y.to(device)
    optimizer.zero_grad()

    with torch.amp.autocast('cuda'):  # 自动混合精度
        output = model(x)
        loss = criterion(output, y)

    scaler.scale(loss).backward()     # 梯度缩放
    scaler.step(optimizer)            # 更新参数（自动 unscale）
    scaler.update()                   # 更新 scaler
```

writer = SummaryWriter('runs/experiment_1')

for epoch in range(num_epochs):
    train_loss, train_acc = train_epoch(...)
    val_loss, val_acc = val_epoch(...)

    writer.add_scalar('Loss/train', train_loss, epoch)
    writer.add_scalar('Loss/val', val_loss, epoch)
    writer.add_scalar('Acc/train', train_acc, epoch)
    writer.add_scalar('Acc/val', val_acc, epoch)
