```python
import numpy as np
import matplotlib.pyplot as plt
import os
from scipy import signal

# --- 1. Mac 环境字体配置 (陈教授专用配置) ---
plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'Heiti TC', 'sans-serif']
plt.rcParams['axes.unicode_minus'] = False

# --- 2. 信号处理逻辑：脉冲响应不变法设计 ---

# 模拟滤波器参数: Ha(s) = 1 / (s + 1)
# 截止频率 Omega_c = 1 rad/s
A = 1.0
pole_s = -1.0 

# 采样参数
T = 0.5  # 采样周期 (秒)
fs = 1/T # 采样频率

# 模拟频率轴 (用于绘图)
Omega = np.linspace(0, np.pi/T, 500) # 0 到 奈奎斯特频率
# 模拟滤波器频率响应 |Ha(jOmega)|
Ha_freq_resp = np.abs(1 / (1j * Omega - pole_s))

# --- 脉冲响应不变法转换 ---
# 极点映射: z_p = exp(s_p * T)
pole_z = np.exp(pole_s * T)

# 数字滤波器系数 (引入T进行增益归一化)
# H(z) = T / (1 - pole_z * z^-1)
# 分子 b = [T], 分母 a = [1, -pole_z]
b_digital = [T]
a_digital = [1, -pole_z]

# 数字频率轴 omega (0 到 pi)
omega = Omega * T 
# 数字滤波器频率响应
w, H_digital = signal.freqz(b_digital, a_digital, worN=omega)

# --- 3. 自动保存与显示 (动态路径配置) ---
filename = "脉冲响应不变法_频率响应对比.png"
current_dir = os.getcwd()
full_path = os.path.join(current_dir, filename)

try:
    if os.path.exists(full_path):
        os.remove(full_path) 
        
    plt.figure(figsize=(10, 6))
    
    # 绘制模拟滤波器响应
    plt.plot(Omega, Ha_freq_resp, 'b--', linewidth=2, label=r'模拟滤波器 $|H_a(j\Omega)|$')
    
    # 绘制数字滤波器响应
    plt.plot(Omega, np.abs(H_digital), 'r-', linewidth=1.5, label=r'数字滤波器 $|H(e^{j\omega})|$ (映射后)')
    
    plt.title(f'脉冲响应不变法：模拟与数字频率响应对比 (T={T}s)', fontsize=14)
    plt.xlabel(r'模拟角频率 $\Omega$ (rad/s)', fontsize=12)
    plt.ylabel('幅度', fontsize=12)
    plt.legend(fontsize=12)
    plt.grid(True, which='both', linestyle='--', alpha=0.6)
    
    # 标注混叠效应的提示
    plt.text(2, 0.6, "注意：高频处因混叠\n两者略有差异", fontsize=10, color='gray', bbox=dict(facecolor='white', alpha=0.8))

    plt.savefig(full_path, dpi=300, bbox_inches='tight')
    plt.close() 
    
    print(f"![[{filename}]]") 
    print(f"转换完成。模拟极点 s={pole_s} -> 数字极点 z={pole_z:.4f}")

except Exception as e:
    print(f"Error: {e}")
```
