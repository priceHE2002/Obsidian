```python
import numpy as np
import matplotlib.pyplot as plt
import os

# --- 1. Mac 环境字体配置 (陈教授专用配置) ---
plt.rcParams['font.sans-serif'] = ['Arial Unicode MS', 'Heiti TC', 'sans-serif']
plt.rcParams['axes.unicode_minus'] = False

# --- 2. 信号处理逻辑 ---

# 定义模拟信号: 频率 f = 1Hz 的正弦波
# x(t) = sin(2*pi*t)
f_sig = 1.0 

# 为了画出"连续"的曲线，我们需要一个极高的采样率来模拟
t_continuous = np.linspace(0, 2, 1000) # 0到2秒
x_continuous = np.sin(2 * np.pi * f_sig * t_continuous)

# --- 情况 A: 低采样频率 (大间隔) ---
fs_low = 5.0 # Hz
T_low = 1 / fs_low # 0.2 秒
# 生成采样点时间轴
k_low = np.arange(0, 2 + T_low, T_low)
x_low = np.sin(2 * np.pi * f_sig * k_low)

# --- 情况 B: 高采样频率 (小间隔) ---
fs_high = 20.0 # Hz
T_high = 1 / fs_high # 0.05 秒
# 生成采样点时间轴
k_high = np.arange(0, 2 + T_high, T_high)
x_high = np.sin(2 * np.pi * f_sig * k_high)

# --- 3. 绘图 ---
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 8))

# 子图1：低采样率
ax1.plot(t_continuous, x_continuous, 'k--', label='连续信号 $x(t)$', alpha=0.5)
ax1.stem(k_low, x_low, linefmt='b-', markerfmt='bo', basefmt=" ", label=f'离散采样 $x[k]$')
ax1.set_title(f'情况A: 采样频率 $f_{{sam}}={fs_low}$ Hz, 采样间隔 $T={T_low}$ s', fontsize=12)
ax1.set_ylabel('幅度')
ax1.grid(True, alpha=0.3)
ax1.legend()
# 标注间隔
ax1.annotate(f'间隔 T={T_low}s', xy=(0.2, 0.95), xytext=(0.4, 0.95),
             arrowprops=dict(arrowstyle='<->', color='red'))

# 子图2：高采样率
ax2.plot(t_continuous, x_continuous, 'k--', label='连续信号 $x(t)$', alpha=0.5)
ax2.stem(k_high, x_high, linefmt='g-', markerfmt='go', basefmt=" ", label=f'离散采样 $x[k]$')
ax2.set_title(f'情况B: 采样频率 $f_{{sam}}={fs_high}$ Hz, 采样间隔 $T={T_high}$ s', fontsize=12)
ax2.set_xlabel('时间 (s)')
ax2.set_ylabel('幅度')
ax2.grid(True, alpha=0.3)
ax2.legend()
# 标注间隔
ax2.annotate(f'间隔 T={T_high}s', xy=(0.2, 0.95), xytext=(0.25, 0.95),
             arrowprops=dict(arrowstyle='<->', color='red'))

plt.tight_layout()

# --- 4. 自动保存与显示 (固定配置) ---
filename = "采样频率与采样间隔的关系演示.png"
vault_path = "/Users/heyuhang/Documents/Obsidian Vault"
full_path = os.path.join(vault_path, filename)

try:
    if os.path.exists(full_path):
        os.remove(full_path) # 覆盖旧图
    plt.savefig(full_path, dpi=300, bbox_inches='tight')
    plt.close() # 释放内存
    print(f"![[{filename}]]") # 触发 Obsidian 显示
except Exception as e:
    print(f"Error: {e}")
```
