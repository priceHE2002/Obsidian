# 1. 张量操作

PyTorch 的核心数据结构是 Tensor，可以理解为能在 GPU 上运行的多维数组。它和 NumPy 的 ndarray 非常相似，但增加了自动微分和 GPU 加速能力。

## 1.1 创建张量

```python
import torch
import numpy as np

# 从列表创建
a = torch.tensor([1, 2, 3])
print(a)  # tensor([1, 2, 3])

# 全零 / 全一 / 随机
zeros = torch.zeros(3, 4)          # shape (3, 4)
ones = torch.ones(2, 3)            # shape (2, 3)
rand = torch.rand(2, 3)            # [0, 1) 均匀分布
randn = torch.randn(2, 3)          # 标准正态分布 N(0, 1)
randint = torch.randint(0, 10, (3, 4))  # [0, 10) 整数

# 指定值 / 等差
full = torch.full((2, 3), 7.0)     # 全部填 7
arange = torch.arange(0, 10, 2)    # tensor([0, 2, 4, 6, 8])
linspace = torch.linspace(0, 1, 5) # tensor([0.00, 0.25, 0.50, 0.75, 1.00])

# 单位矩阵
eye = torch.eye(4)                 # 4x4 单位矩阵

# 从 NumPy 互转（共享内存）
np_arr = np.array([[1, 2], [3, 4]])
t_from_np = torch.from_numpy(np_arr)  # 共享内存
t_to_np = t_from_np.numpy()           # tensor → numpy
```

## 1.2 张量属性

```python
x = torch.randn(2, 3, 4)

print(x.shape)      # torch.Size([2, 3, 4])
print(x.size())     # 同上
print(x.dtype)      # torch.float32
print(x.device)     # cpu
print(x.ndim)       # 3（维度数）
print(x.numel())    # 24（元素总数）
```

**常用 dtype**：`torch.float32`（默认）、`torch.float16`、`torch.int64`、`torch.bool`。

创建时指定类型：`torch.tensor([1,2], dtype=torch.float32)`。

## 1.3 索引与切片

```python
x = torch.randn(3, 4, 5)

print(x[0].shape)         # torch.Size([4, 5]) — 取第一个 batch
print(x[0, :, :].shape)   # 等价
print(x[:, 1, :].shape)   # torch.Size([3, 5]) — 所有 batch 的第 1 行
print(x[..., -1].shape)   # torch.Size([3, 4]) — 取最后一个元素
print(x[:2, :2, :2].shape)# torch.Size([2, 2, 3]) — 范围切片
```

## 1.4 形状变换

```python
x = torch.randn(4, 4)

# view：共享内存，要求连续
y = x.view(16)            # 展平
y = x.view(-1, 2)         # 自动推断第一维：8x2

# reshape：会处理不连续情况（更安全）
y = x.reshape(2, 8)

# 增删维度
y = x.unsqueeze(0)        # shape: (1, 4, 4)
y = x.unsqueeze(-1)       # shape: (4, 4, 1)
y = y.squeeze()           # 去掉所有大小为 1 的维度

# 转置与 permute
y = x.t()                 # 二维转置
y = x.permute(1, 0)       # 任意维度交换
```

## 1.5 张量运算

```python
a = torch.randn(2, 3)
b = torch.randn(2, 3)

# 逐元素运算
c = a + b                 # 加法
c = a - b                 # 减法
c = a * b                 # 逐元素乘法
c = a / b                 # 除法
c = torch.matmul(a, b.t())# 矩阵乘法 (2,3) x (3,2) → (2,2)

# 简写形式
c = a @ b.t()             # 矩阵乘法（推荐写法）

# 广播
a = torch.randn(2, 3)
b = torch.randn(3)        # 自动广播为 (2, 3)
c = a + b

# 聚合
a.sum()                   # 总和
a.mean(dim=0)             # 沿第 0 维求均值
a.max()                   # 最大值
a.argmax(dim=1)           # 沿第 1 维取最大值的下标
```

## 1.6 设备切换（CPU / GPU）

```python
# 检查 GPU 是否可用
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

# 创建时指定设备
x = torch.randn(3, 4, device=device)

# 模型和数据搬到 GPU
x = x.to(device)
x = x.cuda()              # 简写（不推荐：如无 GPU 会报错）

# 从 GPU 搬回 CPU
x = x.cpu()
```

## 1.7 内存与连续性

```python
# clone：深拷贝
y = x.clone()

# detach：从计算图分离，返回新张量（不再跟踪梯度）
y = x.detach()

# contiguous：确保内存连续（view 前常用）
x = x.contiguous()
```
